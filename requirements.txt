# arkiv — Media Asset Manager
# Core
# Upper bounds on the libraries that gate the API/data contract so a future
# major (e.g. chromadb 2.x changing the collection API, pydantic 3.x) can't
# silently break a fresh install. Bump deliberately after testing.
fastapi>=0.100.0
uvicorn>=0.20.0
pydantic>=2.0,<3
chromadb>=1.0,<2
requests>=2.28
xxhash>=2.0.0
nanoid>=2.0.0
# (pandas removed — it was an unused ~70 MB dependency, imported nowhere.)

# Media processing
pillow>=9.0
soundfile>=0.12

# Resource-aware pipeline (Phase 11.5) — soft dependency: arkiv runs without it
# (probe degrades the memory fields to None), but installing it enables
# backpressure and the `arkiv status` memory readout.
psutil>=5.9

# Folder watcher (Phase 11.2) — soft dependency: watch.py falls back to a
# polling scan when watchdog is absent (e.g. the NAS), so this just upgrades the
# watcher to event-driven FSEvents/inotify where available.
watchdog>=3.0

# Speech pre-processing
silero-vad>=6.0
whisper-guard>=0.3   # 4-layer anti-hallucination text filter (was vendored; now PyPI)

# ── Platform-specific Whisper (install ONE) ──────────────
#
# macOS Apple Silicon (recommended):
#   pip install mlx-whisper
#   Uses Metal GPU — fastest on Apple Silicon
#
# NVIDIA GPU (CUDA):
#   pip install faster-whisper torch
#   Requires CUDA toolkit; torch enables GPU acceleration
#
# CPU only (Docker default):
#   pip install faster-whisper
#   No GPU needed, slower but works everywhere
#
faster-whisper>=1.0;platform_system!="Darwin"
mlx-whisper>=0.4;platform_system=="Darwin"

# ── MCP server (Phase 14) — optional ─────────────────────
# Only needed to run the MCP server (`python mcp_server.py`), which exposes
# arkiv's search/metadata to MCP clients (Claude, OpenClaw). The HTTP API and
# CLI work without it. Gated on Python 3.10+ — the MCP SDK requires it, and arkiv
# still supports 3.9 (NAS), where an unconditional entry would break pip install.
mcp>=1.2; python_version >= "3.10"
